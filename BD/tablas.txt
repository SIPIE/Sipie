DROP TABLE profesional;
DROP TABLE participante;

CREATE TABLE profesional(rut varchar(20) PRIMARY KEY, pass varchar(20) NOT NULL, nombres varchar(120) NOT NULL, apellidos varchar(120) NOT NULL, telefono varchar(20) NOT NULL, email varchar(120) NOT NULL, direccion varchar(255) NOT NULL);

CREATE TABLE participante(rut varchar(20) PRIMARY KEY, nombres varchar(120) NOT NULL, apellidos varchar(120) NOT NULL, telefono varchar(20) NOT NULL, email varchar(120) NOT NULL, direccion varchar(255) NOT NULL, sexo varchar(20) NOT NULL, tipo varchar(20) NOT NULL, fechanacimiento varchar(20) NOT NULL, lateralidad varchar(20) NOT NULL);
