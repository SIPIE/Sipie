DROP TABLE profesional;
DROP TABLE participante;

CREATE TABLE profesional(id integer AUTO_INCREMENT PRIMARY key,rut varchar(20) NOT NULL unique, Codigo Mineduc varchar(20) NOT NULL unique,pass varchar(20) NOT NULL, nombres varchar(120) NOT NULL, apellidos varchar(120) NOT NULL, telefono varchar(20) NOT NULL, email varchar(120) NOT NULL, direccion varchar(255) NOT NULL);

CREATE TABLE participante(id integer AUTO_INCREMENT PRIMARY key,rut varchar(20) NOT NULL unique, nombres varchar(120) NOT NULL, apellidos varchar(120) NOT NULL, telefono varchar(20) NOT NULL, email varchar(120) NOT NULL, direccion varchar(255) NOT NULL, sexo varchar(20) NOT NULL, tipo varchar(20) NOT NULL, fechanacimiento varchar(20) NOT NULL, lateralidad varchar(20) NOT NULL);
